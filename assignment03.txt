SELECT customer.customer_name , loan.loan_number
FROM customer
JOIN borrower ON customer.customer_id = borrower.customer_id
JOIN loan ON borrower.loan_number = loan.loan_number WHERE loan.branch_name="Downtown";


SELECT DISTINCT customer.customer_name
FROM customer
JOIN borrower ON customer.customer_id=borrower.customer_id;


SELECT DISTINCT customer.customer_name
FROM customer
LEFT JOIN borrower ON customer.customer_id = borrower.customer_id
WHERE borrower.loan_number IS NULL;


SELECT c1.customer_name AS Customer1 , c2.customer_name AS Customer2 , c1.customer_city
FROM customer AS c1 
JOIN customer AS c2 ON c1.customer_city=c2.customer_city 
AND c1.customer_id < c2.customer_id;


SELECT branch_name, ROUND(SUM(balance * 0.04), 2) AS Total_Interest
FROM account
GROUP BY branch_name;


SELECT account.account_number, customer.customer_city
FROM account
JOIN depositor ON account.account_number=depositor.account_number
JOIN customer ON depositor.customer_id=customer.customer_id
WHERE (customer.customer_city,account.balance) IN (
    SELECT customer.customer_city , MAX(account.balance)
    FROM account
    JOIN depositor ON account.account_number=depositor.account_number
    JOIN customer ON depositor.customer_id=customer.customer_id
    GROUP BY customer.customer_city
);

SELECT loan.loan_number, loan.amount, customer.customer_name
FROM loan
JOIN borrower ON loan.loan_number=borrower.loan_number
JOIN customer ON borrower.customer_id=customer.customer_id
WHERE (loan.amount,loan.loan_number,customer.customer_name) IN (
    SELECT loan.amount 
    FROM loan 
    ORDER BY loan.amount ASC
    LIMIT 5
);
